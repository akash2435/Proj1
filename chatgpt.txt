    <div class="col-md-3">

            {

            <option value="">All</option>
            @foreach (var priority in (ViewBag.PriorityList as SelectList).Items)
            {
                var item = priority as SelectListItem;
                <option value="@item.Value" @(Request.Query["priorityFilter"] == item.Value ? "selected" : "")>
                    @item.Text
                </option>
            }
        </select>
    </div>

    <div class="col-md-3 align-self-end">
        <button type="submit" class="btn btn-primary">Apply Filter</button>
        <a href="@Url.Action("Index", "TaskItems")" class="btn btn-secondary">Reset</a>

            <th>@Html.DisplayNameFor(model => model.Status)</th>
            <th>@Html.DisplayNameFor(model => model.DueDate)</th>

        {
            <tr>
                <td>@Html.DisplayFor(modelItem => item.Title)</td>

                <td>
                    <span class="badge bg-@GetPriorityClass(item.Priority)">@item.Priority</span>
                </td>
                <td>

                </td>
            </tr>
        }
    </tbody>

@functions {
    public string GetPriorityClass(string priority)
    {
        return priority switch
        {
            "High" => "danger",
            "Medium" => "warning",
            "Low" => "success",
            _ => "secondary"
        };
    }

    public string GetStatusClass(string status)
    {
        return status switch
        {
            "Completed" => "success",
            "InProgress" => "primary",
            "Pending" => "secondary",
            _ => "light"
        };
    }
}

@functions {
    public string GetPriorityClass(string priority)
    {
        return priority switch
        {
            "High" => "danger",
            "Medium" => "warning",
            "Low" => "success",
            _ => "secondary"
        };
    }

    public string GetStatusClass(string status)
    {
        return status switch
        {
            "Completed" => "success",
            "InProgress" => "primary",
            "Pending" => "secondary",
            _ => "light"
        };
    }
}

        function formatBillStyle(data) {
            return Object.entries(data).map(([key, value]) => `
                <div class='bill-style'>
                    <span>${key}</span>
                    <span><strong>${value}</strong></span>
                </div>
            `).join('');
        }

        function formatBulletPoints(items) {
            return `<ul>` + items.map(item => `<li>${item}</li>`).join('') + `</ul>`;
        }

        function generateReport(properties) {
            let html = '';
            properties.forEach(property => {
                html += `
                    <div class="property">
                        <h3>${property.title}</h3>
                        <p><strong>Owner:</strong> ${property.owner}</p>
                        <p><strong>Address:</strong> ${property.address}</p>
                        <p><strong>Price:</strong> ${property.price}</p>
                        <p><strong>Size:</strong> ${property.size}</p>
                        <p><strong>Bedrooms:</strong> ${property.bedrooms}</p>
                        <p><strong>Bathrooms:</strong> ${property.bathrooms}</p>
                        <p><strong>Description:</strong> ${property.description}</p>

                        <div class="grid-container">
                            <div class="grid-item"><h4>Deed Tax Breakdown</h4>${formatBillStyle(property.deedTaxBreakdown)}</div>
                            <div class="grid-item"><h4>Mortgage</h4>${formatBillStyle(property.mortgage)}</div>
                            <div class="grid-item"><h4>Mortgage Tax Breakdown</h4>${formatBillStyle(property.mortgageTaxBreakdown)}</div>
                            <div class="grid-item"><h4>Releases</h4>${formatBillStyle(property.releases)}</div>

                    </div>
                `;
            });
            document.getElementById("report").innerHTML = html;
        }
